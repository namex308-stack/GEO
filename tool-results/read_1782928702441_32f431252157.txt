   456→/** Format a string with {placeholder} substitution. */
   457→function format(str: string, params?: Record<string, string | number>): string {
   458→  if (!params) return str;
   459→  return str.replace(/\{(\w+)\}/g, (_, key) => String(params[key] ?? ""));

   462→/**
   463→ * Hook to get the translation function.
   464→ * Usage:
   465→ *   const t = useT();
   466→ *   t("nav.dashboard")           // → "Dashboard" or "لوحة التحكم"
   467→ *   t("dashboard.welcome", { name: "Youssef" })  // → "Welcome back, Youssef 👋"
   468→ */
